        <div class="update-time" id="time">Loading...</div>
    </div>

    <script>
        // 替换为你第一步获得的真实 Web App URL
        const webAppUrl = 'https://script.google.com/macros/s/AKfycbx-zDdX99BrhEg5TLLjGnHnemg-Xn5dV9KfxsBRN6mWXYLzidhpC9i7XdXJByfOf-VOKw/exec';

        async function updateCount() {
            try {
                const response = await fetch(webAppUrl);
                const newCount = await response.text();
                document.getElementById('count').textContent = newCount;
                document.getElementById('time').textContent = `Thời gian thu thập dữ liệu：${new Date().toLocaleTimeString()}`;
            } catch (error) {
                console.error('获取数据失败:', error);
                document.getElementById('count').textContent = 'Err';
                document.getElementById('time').textContent = '更新失败';
            }
        }

        // 页面加载后立即更新一次
        updateCount();
        // 然后每10秒自动更新一次（可根据需要调整间隔）
        setInterval(updateCount, 10000);
    </script>
